const getTargets = async (filters: TargetFilters) => {
  let query = supabase
    .from('targets')
    .select(`
      *,
      projects (name)
    `)
    .order('created_at', { ascending: false });

  if (filters.projectId && filters.projectId !== 'all') {
    query = query.eq('project_id', filters.projectId);
  }

  if (filters.industry && filters.industry !== 'all') {
    query = query.eq('industry', filters.industry);
  }

  if (filters.revenueRange) {
    const [min, max] = getRevenueRange(filters.revenueRange);
    if (min) query = query.gte('estimated_revenue', min);
    if (max) query = query.lte('estimated_revenue', max);
  }

  if (!filters.showExcluded) {
    query = query.eq('is_excluded', false);
  }

  if (filters.search) {
    query = query.ilike('company_name', `%${filters.search}%`);
  }

  return query.range(filters.offset, filters.offset + filters.limit - 1);
};
